大数乘法

  • ACM
  • C++
  • 学习笔记
学习

求两个不超过50位的非负数的积。

代码如下：

#include<iostream>
#include<string>
using namespace std;

int main()
{
    string s1,s2;
    int a[150];//结果的最大位数为两乘数位数的和 
    while(cin>>s1>>s2)
    {
        for(int i=0;i<150;i++)//初始化结果数组 
            a[i]=0;

        if(s1.length()>s2.length())//判断字符串长度，并在短串前面补上0,让两串长度相等 
        {
            string s;
            for(int i=0;i<s1.length()-s2.length();i++)
                s+="0";
            s2=s+s2;
        }
        else
        {
            string s;
            for(int i=0;i<s2.length()-s1.length();i++)
                s+="0";
            s1=s+s1;
        }

        string t;
        t.assign(s1.rbegin(),s1.rend()); 
        s1=t;
        t.assign(s2.rbegin(),s2.rend());//将两个整数串逆置 
        s2=t;

        int d=0;//初始化进位 
        for(int i=0;i<s2.length();i++)//乘法运算，s1的i位与s2的j位相乘得出结果数组的第[i+j]位 
            for(int j=0;j<s2.length();j++)
                a[i+j]+=(s1[i]-'0')*(s2[j]-'0');

    //    for(int i=0;i<2*s2.length();i++)
    //    cout<<a[i]<<" ";
        for(int i=0;i<2*s2.length();i++)//处理进位情况 
        {
            a[i]+=d;//本位加上上一位的进位
            if(a[i]>=10)//当该位大于等于10时，只保留该位的个位 ，向下一位的进位d为十位
            {
                d=a[i]/10;
                a[i]%=10;
            }
            else if(a[i]<10)//当该位小于10是，向高位的进位为0 
                    d=0;
        }
        int mark=0;//高位0标记 
        for(int i=2*s2.length()-1;i>=0;i--)//逆序输出结果 
        {
            if(a[i]==0 && mark==0)//从高位向低位依次输出，高位连续的0不显示 
                continue;
            if(a[i]!=0)
                mark=1;
            cout<<a[i];
        }
        cout<<endl;
    //    for(int i=0;i<2*s2.length();i++)
    //        cout<<a[i]<<" ";
    }
    return 0;
}
